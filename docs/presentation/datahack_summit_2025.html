Get Started in 60 Seconds

# 1. Clone and setup
git clone https://github.com/Leezekun/MassGen
cd MassGen && pip install uv && uv venv

# 2. Configure API keys
cp .env.example .env # Add your API keys

# 3. Run single agent (quick test)
uv run python -m massgen.cli --model gemini-2.5-flash "Compare renewable energy approaches"

# 4. Run multi-agent collaboration
uv run python -m massgen.cli --config three_agents_default.yaml "Complex research question"
